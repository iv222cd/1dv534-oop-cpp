@startuml

participant Application
participant File
participant Values
participant Menu

activate Application
Application -> File : Create File Object
activate File
Application -> File : Read values from File
File -> Values : Store values
deactivate File
activate Values
Application -> Values : Calculate statistics

activate Menu

loop until Quit Choice given.

    Application -> Menu : Print Menu
    Application -> Menu : Read Menu Coice
    
    Application -> Values : Get values and/or statistics
    Application -> Application : Print requested values and/or statistics

end

deactivate Values
deactivate Menu

Application -> Application : Terminate Program

deactivate Application

@enduml
